```python
import json


def safe_json_load(raw, default=None):
    """Safely deserialize a JSON string from a SQLite TEXT column.

    Returns default if raw is None, empty, non-string, or invalid JSON.
    """
    if not isinstance(raw, str) or not raw:
        return default
    try:
        return json.loads(raw)
    except (json.JSONDecodeError, ValueError):
        return default
```

```html
    <!-- Re-fetch table when a score changes (triggered by rescore/paste-jd routes) -->
    <div hx-get="/jobs/table"
         hx-trigger="scoreUpdated from:body"
         hx-target="#job-table-body"
```

This listens for the `scoreUpdated` event on `<body>`, then re-fetches the table. Because the div is inside `<form id="filter-form">`, HTMX automatically includes the form's filter/sort values in the request. The table re-renders in the correct sort order.

The Archive button at `_row_expanded.html:191-197` posts to the status route but has no `hx-target` or `hx-swap`. The status route returns `_status_cell.html` (a `<select>`), so we target the compact row's status cell for the swap, then trigger a table re-fetch to handle filtering/reordering.

Replace the Archive button with:
```html
        <button type="button"
                hx-post="/jobs/{{ job.dedup_key | urlencode }}/status"
                hx-vals='{"pipeline_status": "archived"}'
                hx-target="#status-cell-{{ job.dedup_key | urlencode }}"
                hx-swap="innerHTML"
                hx-confirm="Archive this job?"
                hx-disable-elt="this"
                hx-on::after-request="htmx.trigger(document.body, 'scoreUpdated')"
                class="px-3 py-1 bg-slate-700 hover:bg-slate-600 text-slate-300 rounded transition-colors disabled:opacity-50">
          Archive
          <span class="htmx-indicator ml-1 inline-flex items-center gap-1">

          </span>
        </button>
```

How this works:
- `hx-target` targets the status cell `<td>` in the compact row (ID set at `_row.html:35`)
- `hx-swap="innerHTML"` replaces the cell's content with the updated `<select>` from the status route
- `hx-on::after-request` fires the `scoreUpdated` event, which triggers the table re-fetch listener (from Task 3). If the user is filtering by status, the archived job disappears; otherwise its status updates visibly.

```python
@jobs_bp.route("/<path:dedup_key>/save-jd", methods=["POST"], strict_slashes=False)
def save_jd(dedup_key: str):
    """HTMX POST -- save pasted JD text without triggering AI scoring.

    Must check HX-Request header per architecture rules. Non-HTMX requests
    redirect to the job board page.
    """
    if not request.headers.get("HX-Request"):
        return redirect(url_for("jobs.index"))

    db_path = current_app.config["DB_PATH"]
    conn = get_db(db_path)

    ctx = load_job_context(conn, dedup_key)
    if ctx is None:
        return "", 404

    jd_text = request.form.get("jd_text", "").strip()
    if not jd_text:
        return render_template(
            "jobs/_row_expanded.html",
            job=ctx["job"],
            pipeline_statuses=PIPELINE_STATUSES,
            error="Please provide a job description.",
            resume_history=ctx["resume_history"],
            prep_row=ctx["prep_row"],
            drive_status=get_drive_status(current_app.config.get("JF_CONFIG", {})),
        )

    conn.execute(
        "UPDATE jobs SET jd_full = ? WHERE dedup_key = ?",
        (jd_text, dedup_key),
    )
    conn.commit()

    ctx = load_job_context(conn, dedup_key)
    return render_template(
        "jobs/_row_expanded.html",
        job=ctx["job"],
        pipeline_statuses=PIPELINE_STATUSES,
        resume_history=ctx["resume_history"],
        prep_row=ctx["prep_row"],
        drive_status=get_drive_status(current_app.config.get("JF_CONFIG", {})),
    )
```
