```python
def _parse_job_card(link_tag, email_date: Optional[datetime]) -> Optional[Job]:
    """Parse a single job card from a Glassdoor alert.

    Tries class-based extraction first (legacy gd-* classes), then falls back
    to positional extraction (plain span/p tags with no classes).
    """
    href = link_tag.get("href", "")
    source_id = _extract_listing_id(href)

    # --- Strategy 1: Class-based extraction (legacy gd-* CSS classes) ---
    company_el = link_tag.find("span", class_=COMPANY_CLASS)
    title_el = link_tag.find("p", class_=TITLE_CLASS)

    if company_el and title_el:
        company = company_el.get_text(strip=True)
        title = title_el.get_text(strip=True)
        detail_els = link_tag.find_all("p", class_=DETAIL_CLASS)
        location, salary_min, salary_max = _extract_details_from_class(detail_els)
    else:
        # --- Strategy 2: Positional extraction (no CSS classes) ---
        company, title, salary_min, salary_max = _extract_positional(link_tag)
        location = "Unknown"

    if not title:
        return None

    clean_url = f"https://www.glassdoor.com/job-listing/j?jl={source_id}" if source_id else href

    return Job(
        title=title,
        company=company,
        location=location,
        source="glassdoor",
        source_url=clean_url,
        source_id=source_id,
        salary_min=salary_min,
        salary_max=salary_max,
        posted_date=email_date,
    )


def _extract_details_from_class(detail_els):
    """Extract location/salary from class-based detail elements (legacy)."""
    location = "Unknown"
    salary_min = None
    salary_max = None
    for el in detail_els:
        text = el.get_text(strip=True)
        if _looks_like_salary(text):
            salary_min, salary_max = _parse_salary(text)
        elif text and location == "Unknown":
            location = text
    return location, salary_min, salary_max


def _extract_positional(link_tag):
    """Positional extraction: plain span/p tags with no CSS classes.

    Structure:
      <span> Company (first span without class, may have nested spans)
      <p>    Title   (first <p>)
      <p>    Salary  (contains "$") or Age (skip)
    """
    # Company: first direct child <span> — use first nested <span> for clean name
    spans = link_tag.find_all("span", recursive=False)
    if not spans:
        # Try non-recursive first, fall back to any span
        spans = link_tag.find_all("span")

    company = "Unknown"
    if spans:
        # The first span's first nested span is the clean company name
        nested = spans[0].find("span")
        if nested:
            company = nested.get_text(strip=True)
        else:
            company = spans[0].get_text(strip=True)

    # Title and salary: from <p> tags in order
    p_tags = link_tag.find_all("p")
    title = None
    salary_min = None
    salary_max = None

    for p in p_tags:
        text = p.get_text(strip=True)
        if not text:
            continue
        if title is None:
            title = text
        elif _looks_like_salary(text):
            salary_min, salary_max = _parse_salary(text)
            # Stop after salary — remaining <p> tags are age/noise

    return company, title, salary_min, salary_max
```

```python
import re as _re

# Company name suffixes to strip for domain guessing
_COMPANY_SUFFIXES = _re.compile(
    r"\b(inc\.?|corp\.?|llc\.?|ltd\.?|co\.?|company|group|holdings?|"
    r"technologies|solutions|services|international|global)\b",
    _re.IGNORECASE,
)


def _normalize_company_name(name: str) -> str:
    """Normalize company name for domain guessing.

    Strips legal suffixes, removes non-alphanumeric chars, lowercases.
    'Intuit Inc' -> 'intuit', 'Hinge Health' -> 'hingehealth'
    """
    cleaned = _COMPANY_SUFFIXES.sub("", name)
    cleaned = _re.sub(r"[^a-zA-Z0-9]", "", cleaned)
    return cleaned.lower()


def _try_domain_guess(company_name: str) -> str | None:
    """Tier 1.5: Try https://{normalized_name}.com via HEAD + body validation."""
    normalized = _normalize_company_name(company_name)
    if not normalized or len(normalized) < 2:
        return None

    url = f"https://{normalized}.com"
    try:
        head_resp = requests.head(url, allow_redirects=True, timeout=_TIMEOUT, headers=_HEADERS)
        if head_resp.status_code != 200:
            return None

        content_type = head_resp.headers.get("Content-Type", "")
        if "text/html" not in content_type:
            return None

        get_resp = requests.get(url, timeout=_TIMEOUT, headers=_HEADERS)
        body_sample = get_resp.text[:5000].lower()

        for signature in _PARKED_DOMAIN_SIGNATURES:
            if signature in body_sample:
                logger.debug("Domain guess: %s appears parked", url)
                return None

        return head_resp.url

    except Exception as e:
        logger.debug("Domain guess failed for %s: %s", url, e)
        return None
```

```python
_SLUG_SUFFIXES = ("careers", "jobs", "hiring", "team", "work", "talent")


def _slug_variants(slug: str) -> list[str]:
    """Generate slug variants to try as domains.

    Order: original slug, stripped hyphens, stripped common suffixes.
    """
    variants = [slug]

    # Strip hyphens: "hinge-health" -> "hingehealth"
    if "-" in slug:
        variants.append(slug.replace("-", ""))

    # Strip common suffixes: "frontcareers" -> "front"
    slug_lower = slug.lower().replace("-", "")
    for suffix in _SLUG_SUFFIXES:
        if slug_lower.endswith(suffix) and len(slug_lower) > len(suffix):
            base = slug_lower[: -len(suffix)]
            if base not in variants:
                variants.append(base)

    return variants


def _try_slug_heuristic(ats_slug: str) -> str | None:
    """Try slug variants as .com domains via HEAD + body validation."""
    for variant in _slug_variants(ats_slug):
        url = f"https://{variant}.com"
        try:
            head_resp = requests.head(url, allow_redirects=True, timeout=_TIMEOUT, headers=_HEADERS)
            if head_resp.status_code != 200:
                logger.debug("Slug heuristic: %s returned %d", url, head_resp.status_code)
                continue

            content_type = head_resp.headers.get("Content-Type", "")
            if "text/html" not in content_type:
                logger.debug("Slug heuristic: %s non-HTML: %s", url, content_type)
                continue

            get_resp = requests.get(url, timeout=_TIMEOUT, headers=_HEADERS)
            body_sample = get_resp.text[:5000].lower()

            if any(sig in body_sample for sig in _PARKED_DOMAIN_SIGNATURES):
                logger.debug("Slug heuristic: %s appears parked", url)
                continue

            return head_resp.url

        except Exception as e:
            logger.debug("Slug heuristic failed for %s: %s", url, e)
            continue

    return None
```

```python
def _update_sync_phase(conn, session_id: int, phase: str, detail: str) -> None:
    """Update sync session phase and detail."""
    conn.execute(
        "UPDATE sync_sessions SET phase = ?, detail = ? WHERE id = ?",

def _sync_cancelled(conn, session_id: int) -> bool:
    """Check if sync was cancelled."""
    row = conn.execute(
        "SELECT status FROM sync_sessions WHERE id = ?", (session_id,)
    ).fetchone()
    if row and row["status"] == "cancelling":
        conn.execute(
            "UPDATE sync_sessions SET status = 'cancelled', finished_at = ? WHERE id = ?",
            (utc_now_iso(), session_id),
```

```python
        args=(db_path, session_id, config),
        daemon=True,
    )
    t.start()

    return render_template("dashboard/_sync_progress.html", session_id=session_id,
                           phase="starting", detail="Initializing sync...",
                           status="running")


@dashboard_bp.route("/sync/status/<int:session_id>", strict_slashes=False)
def sync_status(session_id):
    """Poll sync progress — returns self-refreshing HTMX fragment."""
    db_path = current_app.config["DB_PATH"]
    conn = get_db(db_path)

    row = conn.execute(
        "SELECT * FROM sync_sessions WHERE id = ?", (session_id,)
    ).fetchone()

    if not row:
        return "", 404

    return render_template("dashboard/_sync_progress.html",
                           session_id=session_id,
                           phase=row["phase"],
                           detail=row["detail"],
                           status=row["status"],
                           jobs_fetched=row["jobs_fetched"],
                           new_jobs=row["new_jobs"],
                           errors=row["errors"],
                           error_msg=row["error_msg"])
```
