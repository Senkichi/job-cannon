```python
import sqlite3
from contextlib import contextmanager

@contextmanager
def standalone_connection(db_path: str):
    """Context manager for background/CLI sqlite3 connections.

    Sets row_factory=Row and WAL mode. NOT for Flask request handlers (use g.db
    via get_db() instead).

    Usage:
        with standalone_connection(db_path) as conn:
            rows = conn.execute("SELECT ...").fetchall()
            conn.commit()
    """
    conn = sqlite3.connect(db_path)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode=WAL")
    try:
        yield conn
    finally:
        conn.close()
```

### Task 17: Batch stale detector updates (23.3)

Replace per-row `update_pipeline_status()` calls with a single batch UPDATE + batch INSERT for pipeline events.

**Files:**
- Modify: `job_finder/web/stale_detector.py`

**Current pattern:**
```python
for row in rows_to_archive:
    update_pipeline_status(conn, row["dedup_key"], "archived",
                          source="stale_detector", evidence="not_seen_30_days")
    archived += 1
```

Each `update_pipeline_status()` call does: 1 SELECT (check current status) + 1 UPDATE (set new status) + 1 INSERT (pipeline_events) + 1 COMMIT = 4 operations per row.

- [ ] **Step 1: Replace with batch operations**

```python
from job_finder.json_utils import utc_now_iso

rows_to_archive = conn.execute(
    "SELECT dedup_key, pipeline_status FROM jobs "
    f"WHERE last_seen < datetime('now', '-{_ARCHIVE_THRESHOLD_DAYS} days') "
    "AND pipeline_status IN ('discovered', 'reviewing')"
).fetchall()

if rows_to_archive:
    now = utc_now_iso()
    _BATCH_SIZE = 900  # Stay under SQLite's 999 variable limit

    # Process in chunks to avoid SQLite variable limit
    for i in range(0, len(rows_to_archive), _BATCH_SIZE):
        chunk = rows_to_archive[i:i + _BATCH_SIZE]
        keys = [row["dedup_key"] for row in chunk]

        # Batch UPDATE
        placeholders = ",".join("?" for _ in keys)
        conn.execute(
            f"UPDATE jobs SET pipeline_status = 'archived' WHERE dedup_key IN ({placeholders})",
            keys,
        )

        # Batch INSERT pipeline events
        conn.executemany(
            """INSERT INTO pipeline_events
                   (job_id, from_status, to_status, timestamp, source, evidence)
               VALUES (?, ?, 'archived', ?, 'stale_detector', 'not_seen_30_days')""",
            [(row["dedup_key"], row["pipeline_status"], now) for row in chunk],
        )

    conn.commit()
    archived = len(rows_to_archive)
```

**Note:** This assumes all rows transition to 'archived'. The original code checked `from_status == new_status` and skipped — but we already filter for `pipeline_status IN ('discovered', 'reviewing')`, so no row is already 'archived'.
