  ```python
  class AnthropicProvider(BaseProvider):
      def __init__(self, client, conn, config):
          # Store the three-arg constructor pattern matching call_claude's needs
          self._client = client
          self._conn = conn
          self._config = config

      def call(self, model, system, messages, output_schema=None, max_tokens=1024, timeout=None) -> ModelResult:
          # Delegate to call_claude — it handles budget gate, tool-choice schema,
          # cost recording, and token extraction internally.
          # DO NOT call cost_gate() or record_cost() — call_claude does it.
          data, cost_usd = call_claude(
              client=self._client,
              model=model,
              system=system,
              messages=messages,
              output_schema=output_schema,
              conn=self._conn,
              config=self._config,
              max_tokens=max_tokens,
              timeout=timeout,
          )
          return ModelResult(
              data=data,
              cost_usd=cost_usd,
              input_tokens=0,    # call_claude records internally; not re-exposed
              output_tokens=0,
              model=model,
              provider="anthropic",
              schema_valid=True,  # call_claude enforces schema via tool-choice
          )
  ```

  Include proper imports, logging, type hints, and module docstring following codebase conventions.
  Run tests again — all must PASS.
