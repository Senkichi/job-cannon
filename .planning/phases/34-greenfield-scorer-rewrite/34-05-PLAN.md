    def _migration_41_drop_legacy_scores(conn):
        """Migration 41: drop legacy score columns after v3.0 rescore convergence.

        Preflight:
            - Backup gate (CONTEXT D-14): abort if most recent
              backup_userdata_*.tar.gz is > 24h old AND GSD_BACKUP_CONFIRMED=1

        Drops:
            - haiku_score, haiku_summary, sonnet_score columns

            - scoring_provider, scoring_model
            - eval_blocks (vestigial; separate sweep)
            - opus_score, score, job_archetype, legitimacy_note

        No inline rollback — recovery is DB restore from backup.
        """
        import glob, os, time

        class MigrationBlockedError(Exception):
            pass

        def _check_backup_recent():
            if os.environ.get("GSD_BACKUP_CONFIRMED") == "1":
                return
            backups = sorted(glob.glob("backup_userdata_*.tar.gz"), reverse=True)
            if not backups:
                raise MigrationBlockedError(
                    "Migration 41 blocked: No backup_userdata_*.tar.gz found. "
                    "Run bash backup_userdata.sh first, or set GSD_BACKUP_CONFIRMED=1 "

                    f"{age_h:.1f}h old (>24h). Run bash backup_userdata.sh, or set "
                    f"GSD_BACKUP_CONFIRMED=1 to override."
                )

        _check_backup_recent()

        cur = conn.cursor()
        cur.execute("DROP INDEX IF EXISTS idx_jobs_haiku_score")
        cur.execute("ALTER TABLE jobs DROP COLUMN haiku_score")
        cur.execute("ALTER TABLE jobs DROP COLUMN haiku_summary")
        cur.execute("ALTER TABLE jobs DROP COLUMN sonnet_score")


    MIGRATIONS.append(_migration_41_drop_legacy_scores)
