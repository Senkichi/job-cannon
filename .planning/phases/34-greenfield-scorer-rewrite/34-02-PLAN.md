    def score_and_persist_job(
        job: dict,
        conn: sqlite3.Connection,
        config: dict,
        client: Any | None = None,
        scorer_fn: Callable[..., ScoringResult] = None,
    ) -> ScoringResult | None:
        """Unified scoring entry point. Dual-writes new columns AND legacy shim
        atomically per CONTEXT D-16.

        - scorer_fn: defaults to score_job. Injection point preserved for tests.
        - New columns: classification, sub_scores_json, fit_analysis (rationale),
          scoring_provider, scoring_model via persist_job_assessment.
        - Legacy shim (REMOVED in Plan 4):
            haiku_score  ← mean(sub_scores.values()) * 20
            sonnet_score ← same value
            haiku_summary ← rationale.strengths[0] or rationale.gaps[0] or ""
          Shim exists only so Plan 3's progressive read-swap has fresh legacy
          columns to read from. Ordering of haiku_score values is preserved
          (monotonic mapping), which is the load-bearing invariant for Plan 3.
        """
        from job_finder.web.job_scorer import score_job as _default_scorer

        if scorer_fn is None:
            scorer_fn = _default_scorer

        dedup_key = job.get("dedup_key")
        result = scorer_fn(job, conn, config, client=client)

        if result.status != "ok" or result.data is None:
            log.info(
                "score_and_persist_job: skip dedup_key=%s status=%s error=%s",
                dedup_key, result.status, result.error,
            )
            return result

        assessment = result.data
        provider = result.provider
        model = _resolve_scoring_model(config, provider)

        # Compute legacy-shim values BEFORE opening the transaction so we can
        # pass them into the unified UPDATE that already writes new columns.
        mean_sub = (
            sum(assessment.sub_scores.values()) / max(len(assessment.sub_scores), 1)
        )
        legacy_numeric = round(mean_sub * 20, 2)
        strengths = assessment.rationale.get("strengths") or []
        gaps = assessment.rationale.get("gaps") or []
        legacy_summary = (strengths[0] if strengths else (gaps[0] if gaps else "")) or ""

        # Atomic UPDATE: new columns AND legacy shim in ONE statement so one
        # commit lands consistent data. This is CONTEXT D-16's atomicity clause.
        cur = conn.cursor()
        cur.execute(
            "SELECT legitimacy_note FROM jobs WHERE dedup_key = ?", (dedup_key,)
        )
        row = cur.fetchone()
        if row is None:
            return result   # no-op on missing key
        legitimacy_note = row[0]
        final_classification = derive_classification(
            assessment.sub_scores, legitimacy_note
        )
        cur.execute(
            """
            UPDATE jobs
               SET classification   = ?,
                   sub_scores_json  = ?,
                   fit_analysis     = ?,
                   scoring_provider = COALESCE(?, scoring_provider),
                   scoring_model    = COALESCE(?, scoring_model),
                   haiku_score      = ?,
                   sonnet_score     = ?,
                   haiku_summary    = ?
             WHERE dedup_key = ?
            """,
            (
                final_classification,
                json.dumps(assessment.sub_scores),
                json.dumps(assessment.rationale),
                provider,
                model,
                legacy_numeric,
                legacy_numeric,
                legacy_summary,
                dedup_key,
            ),
        )
        conn.commit()
        return result


    def _resolve_scoring_model(config: dict, provider: str | None) -> str | None:
        """Pull the active model ID for the scoring tier."""
        providers_cfg = (config.get("providers") or {})
        scoring_cfg = providers_cfg.get("scoring") or {}
        return scoring_cfg.get("model")
    ```

    def run_scoring(
        new_job_keys: list[str],
        config: dict,
        db_path: str,
    ) -> dict:
        """Unified scoring runner — replaces run_haiku_scoring + run_sonnet_evaluation.

        Pre-score liveness gate preserved per CONTEXT D-11. Batch scope matches
        existing run_sonnet_evaluation (same dedup_key list shape).
        """
        from job_finder.web.scoring_orchestrator import score_and_persist_job
        conn = sqlite3.connect(db_path)
        try:
            conn.row_factory = sqlite3.Row
            summary = {
                "scored": 0,
                "classified_apply": 0,
                "classified_consider": 0,
                "classified_skip": 0,
                "classified_reject": 0,
                "skipped_dead": 0,
                "skipped_no_jd": 0,
                "errors": 0,
            }
            for dedup_key in new_job_keys:
                row = conn.execute(
                    "SELECT * FROM jobs WHERE dedup_key = ?", (dedup_key,)
                ).fetchone()
                if not row:
                    continue
                job = dict(row)
                if not check_job_liveness(job, config):
                    summary["skipped_dead"] += 1
                    continue
                result = score_and_persist_job(job, conn, config)
                if result is None or result.status == "skipped":
                    summary["skipped_no_jd"] += 1
                elif result.status == "error":
                    summary["errors"] += 1
                else:
                    summary["scored"] += 1
                    # Re-read classification for the counter (derive_classification
                    # ran inside score_and_persist_job; single DB read here keeps
                    # the counter faithful to what landed on disk).
                    cls = conn.execute(
                        "SELECT classification FROM jobs WHERE dedup_key = ?",
                        (dedup_key,),
                    ).fetchone()
                    if cls and cls[0]:
                        summary[f"classified_{cls[0]}"] += 1
            return summary
        finally:
            conn.close()
    ```
